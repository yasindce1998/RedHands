// pkg/mcp/server.go
package mcp
import (
    "context"
    "encoding/json"
    "net/http"
)
// Every MCP server implements this
type Tool interface {
    Name()        string
    Description() string
    InputSchema() json.RawMessage
    Execute(ctx context.Context, params json.RawMessage) (*ToolResult, error)
}
type Server struct {
    name    string
    version string
    tools   map[string]Tool
}
func NewServer(name, version string) *Server {
    return &Server{
        name:    name,
        version: version,
        tools:   make(map[string]Tool),
    }
}
func (s *Server) Register(t Tool) {
    s.tools[t.Name()] = t
}
func (s *Server) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    var req JSONRPCRequest
    json.NewDecoder(r.Body).Decode(&req)
    switch req.Method {
    case "initialize":
        s.handleInitialize(w, req)
    case "tools/list":
        s.handleToolsList(w, req)
    case "tools/call":
        s.handleToolsCall(w, r.Context(), req)
    }
}
// pkg/mcp/types.go
type JSONRPCRequest struct {
    JSONRPC string          `json:"jsonrpc"`
    ID      interface{}     `json:"id"`
    Method  string          `json:"method"`
    Params  json.RawMessage `json:"params"`
}
type ToolResult struct {
    Content []ContentBlock `json:"content"`
    IsError bool           `json:"isError,omitempty"`
}
type ContentBlock struct {
    Type string `json:"type"` // "text" | "json"
    Text string `json:"text,omitempty"`
    Data any    `json:"data,omitempty"`
}

// servers/nmap/internal/tools/port_scan.go
package tools
import (
    "context"
    "encoding/json"
    "github.com/yourorg/pentest-mcp/pkg/mcp"
    "github.com/yourorg/pentest-mcp/pkg/executor"
)
type PortScanTool struct {
    exec *executor.BinaryExecutor
}
func (t *PortScanTool) Name() string {
    return "nmap_port_scan"
}
func (t *PortScanTool) Description() string {
    return "Scan target host for open ports and running services"
}
func (t *PortScanTool) InputSchema() json.RawMessage {
    return json.RawMessage(`{
        "type": "object",
        "properties": {
            "target": {
                "type": "string",
                "description": "IP address, hostname, or CIDR range"
            },
            "ports": {
                "type": "string",
                "description": "Port range e.g. '80,443' or '1-1000'",
                "default": "1-1000"
            },
            "timing": {
                "type": "integer",
                "description": "Timing template 0-5 (higher = faster)",
                "default": 3
            }
        },
        "required": ["target"]
    }`)
}
func (t *PortScanTool) Execute(
    ctx context.Context,
    params json.RawMessage,
) (*mcp.ToolResult, error) {
    var p struct {
        Target string `json:"target"`
        Ports  string `json:"ports"`
        Timing int    `json:"timing"`
    }
    json.Unmarshal(params, &p)
    // Validate вҖ” NEVER pass raw input to shell
    if err := validateTarget(p.Target); err != nil {
        return mcp.ErrorResult("invalid target: " + err.Error()), nil
    }
    args := []string{
        "-oX", "-",          // XML output to stdout
        "-sV",               // Service detection
        "--open",            // Only open ports
        "-T" + fmt.Sprint(p.Timing),
        "-p", p.Ports,
        p.Target,
    }
    output, err := t.exec.Run(ctx, "nmap", args...)
    if err != nil {
        return mcp.ErrorResult(err.Error()), nil
    }
    // Parse XML вҶ’ clean struct
    result, err := parseNmapXML(output)
    if err != nil {
        return mcp.ErrorResult(err.Error()), nil
    }
    return &mcp.ToolResult{
        Content: []mcp.ContentBlock{
            {Type: "json", Data: result},
        },
    }, nil
}
// servers/nmap/cmd/main.go
func main() {
    exec := executor.New(executor.Config{
        Timeout:    5 * time.Minute,
        MaxMemory:  "512m",
        AllowedBin: []string{"nmap"},
    })
    srv := mcp.NewServer("nmap-mcp", "1.0.0")
    srv.Register(&tools.PortScanTool{exec: exec})
    srv.Register(&tools.ServiceDetectTool{exec: exec})
    srv.Register(&tools.OSDetectTool{exec: exec})
    srv.Register(&tools.NSEScriptTool{exec: exec})
    http.ListenAndServe(":8001", srv)
}

// NSE scripts have wildly different output shapes
// This is the only place in the parser that needs a typed union
type Script struct {
    ID     string
    Output string       // raw text output always present
    Data   ScriptData   // parsed structured form, if we know the script
}
type ScriptData interface {
    scriptData()
}
// pkg/nmap/nse/normalizer.go
// Route to correct handler based on script ID
func Normalize(id, rawOutput string, elem *xmlScriptElem) ScriptData {
    switch {
    case strings.HasPrefix(id, "http-"):
        return parseHTTPScript(id, rawOutput, elem)
    case strings.HasPrefix(id, "ssl-"):
        return parseSSLScript(id, rawOutput, elem)
    case strings.HasPrefix(id, "smb-"):
        return parseSMBScript(id, rawOutput, elem)
    case id == "vuln" || strings.HasPrefix(id, "vuln-"):
        return parseVulnScript(id, rawOutput, elem)
    default:
        return RawScript{Text: rawOutput}  // graceful fallback
    }
}

package mcp
import (
    "context"
    "encoding/json"
    "net/http"
)
type Tool interface {
    Name()        string
    Description() string
    InputSchema() json.RawMessage
    Execute(ctx context.Context, params json.RawMessage) (*ToolResult, error)
}
type Server struct {
    name    string
    version string
    tools   map[string]Tool
    mid     []Middleware   // auth, audit, rate limit chain
}
func NewServer(name, version string) *Server {
    return &Server{
        name:    name,
        version: version,
        tools:   make(map[string]Tool),
    }
}
func (s *Server) Use(m Middleware)  { s.mid = append(s.mid, m) }
func (s *Server) Register(t Tool)   { s.tools[t.Name()] = t }
func (s *Server) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    var req JSONRPCRequest
    if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
        writeError(w, -32700, "parse error", nil)
        return
    }
    handler := s.route(req.Method)
    for i := len(s.mid) - 1; i >= 0; i-- {
        handler = s.mid[i](handler)
    }
    handler(w, r.WithContext(
        context.WithValue(r.Context(), reqKey{}, req),
    ))
}
func (s *Server) route(method string) http.HandlerFunc {
    switch method {
    case "initialize":    return s.handleInitialize
    case "tools/list":    return s.handleToolsList
    case "tools/call":    return s.handleToolsCall
    default:
        return func(w http.ResponseWriter, r *http.Request) {
            writeError(w, -32601, "method not found", nil)
        }
    }
}

package executor
import (
    "bytes"
    "context"
    "fmt"
    "os/exec"
    "strings"
)
type Config struct {
    Timeout    time.Duration
    AllowedBin []string      // strict allowlist
    MaxOutput  int64         // bytes, prevent memory explosion
}
func DefaultConfig() Config {
    return Config{
        Timeout:   5 * time.Minute,
        MaxOutput: 10 * 1024 * 1024, // 10MB
    }
}
type BinaryExecutor struct {
    cfg Config
}
func New(cfg Config) *BinaryExecutor {
    return &BinaryExecutor{cfg: cfg}
}
func (e *BinaryExecutor) Run(
    ctx context.Context,
    bin string,
    args ...string,
) ([]byte, error) {
    // Strict allowlist check вҖ” never skip this
    if !e.isAllowed(bin) {
        return nil, fmt.Errorf("binary not allowed: %s", bin)
    }
    ctx, cancel := context.WithTimeout(ctx, e.cfg.Timeout)
    defer cancel()
    cmd := exec.CommandContext(ctx, bin, args...)
    var stdout, stderr bytes.Buffer
    cmd.Stdout = io.LimitWriter(&stdout, e.cfg.MaxOutput)
    cmd.Stderr = &stderr
    if err := cmd.Run(); err != nil {
        if ctx.Err() == context.DeadlineExceeded {
            return nil, fmt.Errorf("execution timed out after %s", e.cfg.Timeout)
        }
        return nil, fmt.Errorf("execution failed: %w\nstderr: %s", err, stderr.String())
    }
    return stdout.Bytes(), nil
}
func (e *BinaryExecutor) isAllowed(bin string) bool {
    // Empty allowlist = allow all (dev mode only)
    if len(e.cfg.AllowedBin) == 0 {
        return true
    }
    for _, a := range e.cfg.AllowedBin {
        if a == bin {
            return true
        }
    }
    return false
}

package nmap
import (
    "context"
    "encoding/json"
    "fmt"
    "net"
    "github.com/yourorg/pentest-mcp/pkg/executor"
    "github.com/yourorg/pentest-mcp/pkg/mcp"
    "github.com/yourorg/pentest-mcp/pkg/nmap/parser"
    "github.com/yourorg/pentest-mcp/pkg/nmap/query"
)
type PortScanTool struct {
    exec *executor.BinaryExecutor
}
func NewPortScanTool(e *executor.BinaryExecutor) *PortScanTool {
    return &PortScanTool{exec: e}
}
func (t *PortScanTool) Name() string { return "nmap_port_scan" }
func (t *PortScanTool) Description() string {
    return "Scan a target for open ports and services. Returns structured findings."
}
func (t *PortScanTool) InputSchema() json.RawMessage {
    return json.RawMessage(`{
        "type": "object",
        "properties": {
            "target": {
                "type": "string",
                "description": "IP, hostname, or CIDR range to scan"
            },
            "ports": {
                "type": "string",
                "description": "Port range e.g. '80,443' or '1-1000'. Default: top 1000",
                "default": "--top-ports 1000"
            },
            "timing": {
                "type": "integer",
                "description": "Timing template 1-4. Default 3. Use 2 for stealth.",
                "default": 3,
                "minimum": 1,
                "maximum": 4
            }
        },
        "required": ["target"]
    }`)
}
func (t *PortScanTool) Execute(
    ctx context.Context,
    params json.RawMessage,
) (*mcp.ToolResult, error) {
    var p struct {
        Target string `json:"target"`
        Ports  string `json:"ports"`
        Timing int    `json:"timing"`
    }
    // Set defaults
    p.Timing = 3
    p.Ports = "--top-ports 1000"
    if err := json.Unmarshal(params, &p); err != nil {
        return mcp.ErrorResult("invalid params: " + err.Error()), nil
    }
    // Validate target вҖ” never trust LLM input directly
    if err := validateTarget(p.Target); err != nil {
        return mcp.ErrorResult(err.Error()), nil
    }
    args := buildArgs(p.Target, p.Ports, p.Timing)
    raw, err := t.exec.Run(ctx, "nmap", args...)
    if err != nil {
        return mcp.ErrorResult(err.Error()), nil
    }
    run, err := parser.Parse(raw)
    if err != nil {
        return mcp.ErrorResult("parse failed: " + err.Error()), nil
    }
    // Two outputs: summary for agent reasoning, full data for reference
    summary := query.Summarize(run)
    return &mcp.ToolResult{
        Content: []mcp.ContentBlock{
            {Type: "text", Text: formatSummary(summary)},
            {Type: "json", Data: run},
        },
    }, nil
}
func validateTarget(target string) error {
    // Reject empty
    if strings.TrimSpace(target) == "" {
        return fmt.Errorf("target cannot be empty")
    }
    // Reject shell metacharacters
    bad := []string{";", "&", "|", "`", "$", ">", "<", "\n"}
    for _, c := range bad {
        if strings.Contains(target, c) {
            return fmt.Errorf("invalid character in target: %q", c)
        }
    }
    // Must be valid IP, CIDR, or hostname
    if net.ParseIP(target) != nil {
        return nil
    }
    if _, _, err := net.ParseCIDR(target); err == nil {
        return nil
    }
    if isValidHostname(target) {
        return nil
    }
    return fmt.Errorf("target must be a valid IP, CIDR, or hostname")
}
func buildArgs(target, ports string, timing int) []string {
    args := []string{
        "-oX", "-",                       // XML to stdout
        "-sV",                            // Service detection
        "--open",                         // Only open ports
        fmt.Sprintf("-T%d", timing),
    }
    if ports == "--top-ports 1000" {
        args = append(args, "--top-ports", "1000")
    } else {
        args = append(args, "-p", ports)
    }
    args = append(args, target)
    return args
}

-- migrations/001_initial_schema.sql
-- Multi-tenancy foundation
CREATE TABLE organizations (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name        TEXT NOT NULL,
    slug        TEXT NOT NULL UNIQUE,
    plan        TEXT NOT NULL DEFAULT 'enterprise',
    created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE TABLE workspaces (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    org_id          UUID NOT NULL REFERENCES organizations(id),
    name            TEXT NOT NULL,
    slug            TEXT NOT NULL,
    settings        JSONB NOT NULL DEFAULT '{}',
    scan_quota      INTEGER NOT NULL DEFAULT 1000,   -- scans per month
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE(org_id, slug)
);
CREATE TABLE users (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    org_id          UUID NOT NULL REFERENCES organizations(id),
    email           TEXT NOT NULL UNIQUE,
    role            TEXT NOT NULL,                   -- admin|analyst|viewer
    last_login      TIMESTAMPTZ,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE TABLE service_accounts (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    workspace_id    UUID NOT NULL REFERENCES workspaces(id),
    name            TEXT NOT NULL,
    key_hash        TEXT NOT NULL,                   -- bcrypt hash of API key
    permissions     TEXT[] NOT NULL,                 -- tool allowlist
    last_used_at    TIMESTAMPTZ,
    expires_at      TIMESTAMPTZ,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
-- Scan management
CREATE TABLE scans (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    workspace_id    UUID NOT NULL REFERENCES workspaces(id),
    created_by      UUID NOT NULL REFERENCES users(id),
    name            TEXT NOT NULL,
    status          TEXT NOT NULL DEFAULT 'pending', -- pending|running|completed|failed
    scan_type       TEXT NOT NULL,                   -- tool name
    target          TEXT NOT NULL,
    params          JSONB NOT NULL DEFAULT '{}',
    started_at      TIMESTAMPTZ,
    completed_at    TIMESTAMPTZ,
    artifact_path   TEXT,                            -- S3 path to raw output
    error           TEXT,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_scans_workspace    ON scans(workspace_id, created_at DESC);
CREATE INDEX idx_scans_status       ON scans(status) WHERE status IN ('pending','running');
-- Normalized findings
CREATE TABLE findings (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    workspace_id    UUID NOT NULL REFERENCES workspaces(id),
    scan_id         UUID NOT NULL REFERENCES scans(id),
    fingerprint     TEXT NOT NULL,                   -- dedup hash
    title           TEXT NOT NULL,
    description     TEXT,
    severity        TEXT NOT NULL,                   -- critical|high|medium|low|info
    cvss_score      NUMERIC(3,1),
    cve_ids         TEXT[],
    asset           TEXT NOT NULL,                   -- affected host/url
    port            INTEGER,
    service         TEXT,
    tool            TEXT NOT NULL,                   -- which tool found it
    raw_data        JSONB NOT NULL,                  -- tool-specific detail
    status          TEXT NOT NULL DEFAULT 'open',    -- open|acknowledged|false_positive|fixed
    first_seen      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    last_seen       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    ticket_url      TEXT,                            -- Jira/ServiceNow link
    UNIQUE(workspace_id, fingerprint)                -- dedup per workspace
);
CREATE INDEX idx_findings_workspace  ON findings(workspace_id, severity, status);
CREATE INDEX idx_findings_asset      ON findings(asset);
CREATE INDEX idx_findings_cve        ON findings USING GIN(cve_ids);
-- Immutable audit log
CREATE TABLE audit_events (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    org_id          UUID NOT NULL,
    workspace_id    UUID,
    actor_id        UUID NOT NULL,
    actor_type      TEXT NOT NULL,                   -- user|service_account|agent
    actor_ip        INET,
    action          TEXT NOT NULL,                   -- tool_call|scan_create|etc
    resource_type   TEXT,
    resource_id     UUID,
    tool_name       TEXT,
    tool_params     JSONB,                           -- sanitized params
    result_summary  TEXT,
    duration_ms     INTEGER,
    status          TEXT NOT NULL,                   -- success|failure
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
) PARTITION BY RANGE (created_at);                   -- partition by month
-- Audit log is append-only вҖ” enforce with RLS
ALTER TABLE audit_events ENABLE ROW LEVEL SECURITY;
CREATE POLICY audit_insert_only ON audit_events FOR INSERT WITH CHECK (true);
-- No UPDATE, no DELETE policy = immutable
-- Scheduled scans
CREATE TABLE scan_schedules (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    workspace_id    UUID NOT NULL REFERENCES workspaces(id),
    name            TEXT NOT NULL,
    cron_expr       TEXT NOT NULL,                   -- "0 2 * * 1" = Mon 2am
    scan_type       TEXT NOT NULL,
    target          TEXT NOT NULL,
    params          JSONB NOT NULL DEFAULT '{}',
    enabled         BOOLEAN NOT NULL DEFAULT true,
    last_run_at     TIMESTAMPTZ,
    next_run_at     TIMESTAMPTZ NOT NULL,
    created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

type Permission string

    PermScanDelete    Permission = "scan:delete"
    // Finding permissions
    PermFindingRead   Permission = "finding:read"
    PermFindingUpdate Permission = "finding:update"
    PermFindingExport Permission = "finding:export"
    // Admin permissions
    PermWorkspaceAdmin Permission = "workspace:admin"
    PermOrgAdmin       Permission = "org:admin"
)
var RolePermissions = map[string][]Permission{
    "org:admin": {
        PermOrgAdmin, PermWorkspaceAdmin,
        PermToolNmapRun, PermToolNucleiRun,
        PermToolKubeHunterRun, PermToolTrivyRun,
        PermScanCreate, PermScanRead, PermScanDelete,
        PermFindingRead, PermFindingUpdate, PermFindingExport,
    },
    "ws:analyst": {
        PermToolNmapRun, PermToolNucleiRun, PermToolTrivyRun,
        PermScanCreate, PermScanRead,
        PermFindingRead, PermFindingUpdate, PermFindingExport,
    },
    "ws:viewer": {
        PermFindingRead,
        PermScanRead,
    },
}
// Middleware вҖ” every request checked before execution
func RequirePermission(perm Permission) mcp.Middleware {
    return func(next http.HandlerFunc) http.HandlerFunc {
        return func(w http.ResponseWriter, r *http.Request) {
            claims := auth.ClaimsFromContext(r.Context())
            if !claims.HasPermission(perm) {
                writeError(w, 403, "insufficient permissions", map[string]any{
                    "required": perm,
                    "role":     claims.Role,
                })
                return
            }
            next(w, r)
        }
    }
}

// pkg/audit/logger.go
type EventType string
const (
    EventToolCall       EventType = "tool_call"
    EventScanCreate     EventType = "scan_create"
    EventScanComplete   EventType = "scan_complete"
    EventFindingCreate  EventType = "finding_create"
    EventAuthSuccess    EventType = "auth_success"
    EventAuthFailure    EventType = "auth_failure"
    EventPermDenied     EventType = "permission_denied"
)
type AuditEvent struct {
    ID            string            `json:"id"`
    OrgID         string            `json:"org_id"`
    WorkspaceID   string            `json:"workspace_id,omitempty"`
    ActorID       string            `json:"actor_id"`
    ActorType     string            `json:"actor_type"`    // user|service_account|agent
    ActorIP       string            `json:"actor_ip"`
    EventType     EventType         `json:"event_type"`
    ToolName      string            `json:"tool_name,omitempty"`
    ToolParams    map[string]any    `json:"tool_params,omitempty"`  // sanitized
    ResourceType  string            `json:"resource_type,omitempty"`
    ResourceID    string            `json:"resource_id,omitempty"`
    Status        string            `json:"status"`       // success|failure
    DurationMs    int64             `json:"duration_ms,omitempty"`
    ErrorMsg      string            `json:"error,omitempty"`
    Timestamp     time.Time         `json:"timestamp"`
}
type AuditLogger struct {
    db       *postgres.Client
    kafka    *kafka.Producer
    topic    string
}
// Every tool call goes through this вҖ” non-negotiable
func (l *AuditLogger) LogToolCall(
    ctx context.Context,
    tool string,
    params map[string]any,
    duration time.Duration,
    err error,
) {
    claims := auth.ClaimsFromContext(ctx)
    event := AuditEvent{
        ID:          uuid.New().String(),
        OrgID:       claims.OrgID,
        WorkspaceID: claims.WorkspaceID,
        ActorID:     claims.ActorID,
        ActorType:   claims.ActorType,
        ActorIP:     claims.IP,
        EventType:   EventToolCall,
        ToolName:    tool,
        ToolParams:  sanitize(params),   // strip sensitive values
        Status:      statusFrom(err),
        DurationMs:  duration.Milliseconds(),
        Timestamp:   time.Now().UTC(),
    }
    if err != nil {
        event.ErrorMsg = err.Error()
    }
    // Write to DB (queryable) + Kafka (SIEM forwarding)
    go l.db.InsertAuditEvent(context.Background(), event)
    go l.kafka.Publish(l.topic, event)
}
// Sanitize params вҖ” never log credentials, tokens, or PII
func sanitize(params map[string]any) map[string]any {
    sensitive := map[string]bool{
        "password": true, "token": true, "key": true,
        "secret": true, "credential": true,
    }
    out := make(map[string]any, len(params))
    for k, v := range params {
        if sensitive[strings.ToLower(k)] {
            out[k] = "[REDACTED]"
        } else {
            out[k] = v
        }
    }
    return out
}

// pkg/findings/finding.go
// Universal finding format вҖ” all tools output this
type Severity string
const (
    SeverityCritical Severity = "critical"  // CVSS 9.0-10.0
    SeverityHigh     Severity = "high"      // CVSS 7.0-8.9
    SeverityMedium   Severity = "medium"    // CVSS 4.0-6.9
    SeverityLow      Severity = "low"       // CVSS 0.1-3.9
    SeverityInfo     Severity = "info"      // no CVSS
)
type Finding struct {
    ID          string            `json:"id"`
    WorkspaceID string            `json:"workspace_id"`
    ScanID      string            `json:"scan_id"`
    Fingerprint string            `json:"fingerprint"`  // SHA256 of dedup key
    Title       string            `json:"title"`
    Description string            `json:"description"`
    Severity    Severity          `json:"severity"`
    CVSS        float64           `json:"cvss_score,omitempty"`
    CVEIDs      []string          `json:"cve_ids,omitempty"`
    Asset       string            `json:"asset"`        // host or URL
    Port        int               `json:"port,omitempty"`
    Service     string            `json:"service,omitempty"`
    Tool        string            `json:"tool"`
    References  []string          `json:"references,omitempty"`
    Remediation string            `json:"remediation,omitempty"`
    RawData     json.RawMessage   `json:"raw_data"`     // tool-specific
    FirstSeen   time.Time         `json:"first_seen"`
    LastSeen    time.Time         `json:"last_seen"`
    Status      FindingStatus     `json:"status"`
}
// Dedup: same vuln on same asset = update LastSeen, not new finding
func (f *Finding) ComputeFingerprint() string {
    key := fmt.Sprintf("%s:%s:%s:%d",
        f.WorkspaceID, f.Title, f.Asset, f.Port)
    h := sha256.Sum256([]byte(key))
    return hex.EncodeToString(h[:])
}
